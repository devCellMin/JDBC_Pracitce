CREATE TABLE `worldcup`.`player` (
  `idx` INT(11) NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(20) NULL DEFAULT NULL,
  `age` VARCHAR(3) NULL DEFAULT NULL,
  `weight` VARCHAR(3) NULL DEFAULT NULL,
  `position` VARCHAR(20) NULL DEFAULT NULL,
  `backNumber` VARCHAR(3) NULL DEFAULT NULL,
  `hometown` VARCHAR(20) NULL DEFAULT NULL,
  PRIMARY KEY (`idx`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8
COLLATE = utf8_unicode_ci;
